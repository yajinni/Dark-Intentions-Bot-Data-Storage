# Dark Intentions Bot - WCL GraphQL Queries

This file documents all the GraphQL queries used by the bot to interact with the Warcraft Logs API v2.

## 1. Discovery & Rankings

### Character Rankings Search
**Source:** `wcl-discovery.js`
Used to find top-performing logs for comparison.
```graphql
query($encounterID: Int, $difficulty: Int, $className: String, $specName: String, $metric: CharacterRankingMetricType, $filter: String, $page: Int, $partition: Int) {
    worldData {
        encounter(id: $encounterID) {
            characterRankings(difficulty: $difficulty, className: $className, specName: $specName, metric: $metric, filter: $filter, page: $page, partition: $partition)
        }
    }
}
```

### Zone Partitions
**Source:** `wcl-discovery.js`
Used to handle expansion/patch-specific data partitions.
```graphql
query($encounterID: Int) {
    worldData {
        encounter(id: $encounterID) {
            zone {
                partitions {
                    id
                    default
                }
            }
        }
    }
}
```

## 2. Report & Fight Metadata

### Report Overview
**Source:** `bot.js`
Used when a user pastes a log URL to list available boss kills.
```graphql
query($code: String) {
    reportData {
        report(code: $code) {
            title
            fights {
                id
                name
                difficulty
                kill
                bossPercentage
            }
        }
    }
}
```

### Fight Metadata
**Source:** `bot.js`
Used to get exact start/end times and encounter settings.
```graphql
query($code: String, $fightId: [Int]) {
    reportData {
        report(code: $code) {
            startTime
            fights(fightIDs: $fightId) {
                startTime
                endTime
                encounterID
                name
                difficulty
                size
            }
        }
    }
}
```

## 3. Character Analysis

### Hero Talent Detection & Initial Scan
**Source:** `bot.js` / `wcl-discovery.js`
Used to detect the player's Hero Spec by checking for signature ability casts/buffs.
```graphql
query($code: String, $fightId: [Int], $sourceId: Int, $dataType: TableDataType) {
    reportData {
        report(code: $code) {
            table(fightIDs: $fightId, dataType: $dataType, sourceID: $sourceId)
            buffsTable: table(fightIDs: $fightId, dataType: Buffs, sourceID: $sourceId)
            masterData {
                actors {
                    id
                    name
                    subType
                    petOwner
                }
            }
            heroCasts: events(fightIDs: $fightId, dataType: Casts, filterExpression: "[SIGNATURE_FILTER]") { data }
            heroBuffs: events(fightIDs: $fightId, dataType: Buffs, filterExpression: "[SIGNATURE_FILTER]") { data }
            heroDamage: events(fightIDs: $fightId, dataType: DamageDone, filterExpression: "[SIGNATURE_FILTER]") { data }
            heroHealing: events(fightIDs: $fightId, dataType: Healing, filterExpression: "[SIGNATURE_FILTER]") { data }
        }
    }
}
```

### Performance Tables
**Source:** `bot.js`
Used to fetch Damage/Healing, Buffs, and Casts summary tables.
```graphql
query($code: String, $fightId: [Int], $sourceId: Int, $dataType: TableDataType) {
    reportData {
        report(code: $code) {
            mainTable: table(fightIDs: $fightId, dataType: $dataType, sourceID: $sourceId)
            summaryTable: table(fightIDs: $fightId, dataType: Summary)
            buffsTable: table(fightIDs: $fightId, dataType: Buffs, sourceID: $sourceId)
            castsTable: table(fightIDs: $fightId, dataType: Casts, sourceID: $sourceId)
        }
    }
}
```

### Detailed Combat Events (Timeline)
**Source:** `bot.js`
Used for deep analysis of cast sequencing and buff uptimes.
```graphql
query($code: String, $fightId: [Int], $sourceId: Int) {
    reportData {
        report(code: $code) {
            castsEvents: events(fightIDs: $fightId, sourceID: $sourceId, dataType: Casts, limit: 10000) { data }
            buffsEvents: events(fightIDs: $fightId, sourceID: $sourceId, dataType: Buffs, limit: 10000) { data }
            blEvents: events(fightIDs: $fightId, targetID: $sourceId, dataType: Buffs, filterExpression: "ability.id in (2825, 32182, 80353, 390386, 264667)") { data }
        }
    }
}
```

## 4. Rankings & Parses

### Character Parse Data
**Source:** `bot.js`
Used to retrieve the character's performance percentile (rankPercent).
```graphql
query($code: String, $fightIDs: [Int]) {
    reportData {
        report(code: $code) {
            rankings(fightIDs: $fightIDs)
        }
    }
}
```
